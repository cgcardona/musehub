⚡ Agent & MCP Integration

Agents can discover and use this domain through the MuseHub MCP server — no configuration needed.

MCP 
// MCP tool — preferred for AI agents
musehub_get_domain(scoped_id="{{ domain.scoped_id }}")

// REST API — for programmatic access
GET /api/v1/domains/{{ domain.scoped_id }}
GET /api/v1/domains/{{ domain.scoped_id }}/repos

// Human URL (also serves JSON via ?format=json or Accept header)
GET /domains/{{ domain.scoped_id }}?format=json

// MCP resource URI
muse://domains/{{ domain.author_slug }}/{{ domain.slug }}

// Create a repo using this domain
musehub_create_repo(
  name="my-project",
  owner="myuser",
  domain="{{ domain.scoped_id }}"
)
